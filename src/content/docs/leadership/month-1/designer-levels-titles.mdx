<script>
{`
  document.addEventListener('DOMContentLoaded', () => {
    const triggers = document.querySelectorAll('.role-trigger');
    
    // Open first accordion by default
    const firstContent = document.querySelector('.role-content');
    const firstTrigger = document.querySelector('.role-trigger');
    firstContent.classList.add('open');
    firstTrigger.classList.add('open');
    
    triggers.forEach(trigger => {
      trigger.addEventListener('click', () => {
        const content = trigger.nextElementSibling;
        const isOpen = content.classList.contains('open');
        
        // Close all dropdowns
        document.querySelectorAll('.role-content').forEach(content => {
          content.classList.remove('open');
        });
        document.querySelectorAll('.role-trigger').forEach(trigger => {
          trigger.classList.remove('open');
        });
        
        // Toggle clicked dropdown
        if (!isOpen) {
          content.classList.add('open');
          trigger.classList.add('open');
        }
      });
    });
  });
`}
</script>
